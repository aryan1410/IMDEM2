function golf(par, strokes) {
if (strokes == 1) {
document.getElementById("case").innerHTML = "Hole-in-One!";
} else if (strokes <= par - 2) {
document.getElementById("case").innerHTML = "Eagle";
} else if (strokes === par - 1) {
document.getElementById("case").innerHTML = "Birdie";
} else if (strokes === par) {
document.getElementById("case").innerHTML = "Par";
} else if (strokes === par + 1) {
document.getElementById("case").innerHTML = "Bogey";
} else if (strokes === par + 2) {
document.getElementById("case").innerHTML = "Double Bogey";
} else {
document.getElementById("case").innerHTML = "Par";
}
}
golf(5, 4);
//Iteration to find factorial of a number a using for loop
function Fact(a){
let j = 1;
for(let i = 0;i<=a;i++){
j = j*i;
}
document.getElementById("fact").innerHTML = j;
}
Fact(5);
